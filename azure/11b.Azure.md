# 3. Crear el archivo provider.tf

> Configura el provider de Azure
```t
provider "azurerm" {
  features {}
}
```
> [Mas información del recurso azurerm](https://registry.terraform.io/providers/hashicorp/azurerm/latest/docs/guides/features-block)

# 4. Crear el archivo terraform.tf especificando la versión de Azure
```t
terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
      version = "~> 3.0.2"
    }
  }
}
```

## 6. Crear el archivo main.tf

Crear un Storage Account, un contenedor público y un blob con un archivo de contenido

```t
resource "azurerm_resource_group" "demo_rg" {
  name     = "terralabs-resources"
  location = "East US"
}

resource "azurerm_storage_account" "demo_storage" {
  name                     = "terralabsstorage15923" # el nombre debe ser unico, sin guiones ni mayusculas
  resource_group_name      = azurerm_resource_group.demo_rg.name
  location                 = azurerm_resource_group.demo_rg.location
  account_tier             = "Standard" # Standard, Premium
  account_replication_type = "LRS" # LRS, GRS, RA-GRS, ZRS, GZRS, RA-GZRS

  tags = {
    Environment = "Dev"
  }
}

resource "azurerm_storage_container" "demo_container" {
  name                  = "content"
  storage_account_name  = azurerm_storage_account.demo_storage.name
  container_access_type = "blob" # acceso público al blob
}

resource "azurerm_storage_blob" "demo_file" {
  name                   = "demo_azure.png"
  storage_account_name   = azurerm_storage_account.demo_storage.name
  storage_container_name = azurerm_storage_container.demo_container.name
  type                   = "Block"
  source                 = "./assets/demo_azure.png"
  content_type           = "image/png"
}
```

## 7. Crear un output para mostrar el url de la imagen subida a Azure
```t
output "blob_url" {
  value = azurerm_storage_blob.demo_file.url
}
```
